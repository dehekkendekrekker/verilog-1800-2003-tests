====== sequence declaration ======
Created Tuesday 16 December 2025

IEEE 1800-2023 Section A.2.10, 16.8

===== Permutations =====

==== Permutation 1: Basic sequence without arguments ====
sequence s;
	a ##1 b;
endsequence

==== Permutation 2: Sequence with port list ====
sequence s(a, b);
	a ##1 b;
endsequence

==== Permutation 3: Sequence with typed arguments ====
sequence s(logic a, logic b, int delay);
	a ##delay b;
endsequence

==== Permutation 4: Sequence with local variables ====
sequence s;
	logic [7:0] captured;
	(a, captured = data) ##[1:5] (data == captured);
endsequence

==== Permutation 5: Sequence with default arguments ====
sequence s(a, b = 1'b1);
	a ##1 b;
endsequence

==== Permutation 6: Sequence with named end label ====
sequence s;
	a ##1 b;
endsequence : s

==== Permutation 7: Sequence with formal type 'sequence' ====
sequence s(sequence sub_seq);
	a ##1 sub_seq;
endsequence

==== Permutation 8: Sequence with formal type 'untyped' ====
sequence s(untyped expr);
	expr ##1 b;
endsequence
